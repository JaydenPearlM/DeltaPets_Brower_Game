  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT pve_research_stats_pkey PRIMARY KEY (user_id),
  CONSTRAINT pve_research_stats_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.passive_traits (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  key text NOT NULL UNIQUE,
  name text NOT NULL,
  stat_key text NOT NULL CHECK (stat_key = ANY (ARRAY['hp'::text, 'atk'::text, 'def'::text, 'spd'::text, 'magi'::text, 'mana'::text])),
  rarity text NOT NULL CHECK (rarity = ANY (ARRAY['common'::text, 'uncommon'::text, 'rare'::text, 'legendary'::text])),
  description text NOT NULL,
  effect_summary text NOT NULL,
  effects jsonb NOT NULL DEFAULT '{}'::jsonb,
  is_active boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT passive_traits_pkey PRIMARY KEY (id)
);
CREATE TABLE public.mutations (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  key text NOT NULL UNIQUE,
  name text NOT NULL,
  rarity text NOT NULL CHECK (rarity = ANY (ARRAY['common'::text, 'uncommon'::text, 'rare'::text, 'legendary'::text])),
  description text NOT NULL,
  effect_summary text NOT NULL,
  drawback_summary text,
  effects jsonb NOT NULL DEFAULT '{}'::jsonb,
  is_active boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT mutations_pkey PRIMARY KEY (id)
);

  pet_id uuid NOT NULL,
  mutation_id uuid,

  roll_value double precision NOT NULL CHECK (roll_value >= 0::double precision AND roll_value < 1::double precision),
  assigned_at timestamp with time zone NOT NULL DEFAULT now(),
  slot_index smallint NOT NULL DEFAULT 1 CHECK (slot_index >= 1 AND slot_index <= 4),
  CONSTRAINT pet_mutations_pkey PRIMARY KEY (pet_id, slot_index),
  CONSTRAINT pet_mutations_pet_id_fkey FOREIGN KEY (pet_id) REFERENCES public.pets(id),
  CONSTRAINT pet_mutations_mutation_id_fkey FOREIGN KEY (mutation_id) REFERENCES public.mutations(id)
